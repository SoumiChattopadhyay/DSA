#include<stdio.h>
#define INF 99999
#define V 4
void printSolution(int graph[V][V]){
    for(int i=0;i<V;i++){
        for(int j=0;j<V;j++){
            if(graph[i][j]==INF)
                printf("INF ");
            printf("%3d ",graph[i][j]);
        }
        printf("\n");
    }
}
void floydWarshall(int graph[V][V]){
    int distance[V][V];
    for(int i=0;i<V;i++){
        for(int j=0;j<V;j++){
            distance[i][j]=graph[i][j];
        }
    }
    // Floyd warshall algorithm
    for(int k=0;k<V;k++){
        for(int i=0;i<V;i++){
            for(int j=0;j<V;j++){
            if(distance[i][j]>distance[i][k]+distance[k][j]){
                distance[i][j]=distance[i][k]+distance[k][j];
            }
        }
    }
    }
    printSolution(distance);
}
void main(){
    int graph[V][V]=
    {{0,3,INF,7},
    {8,0,2,INF},
    {5,INF,0,1},
    {2,INF,INF,0}};
    floydWarshall(graph);
}
