int minKey(int key[V],int mst[V]){
    int min_val=INT_MAX,min_idx;
    for(int i=0;i<V;i++){
        if(mst[V]==false && key[V]<min_val)
            min_val=key[V];
            min_idx=i;
    }
    return min_idx;
}
