#include<stdio.h>
#include<stdbool.h>
#define MAX 20
int board[MAX][MAX];
void printSolution(int N){
    for(int i=0;i<N;i++){
        for(int j=0;j<N;j++){
            printf("%d ",board[i][j]);
        }
        printf("\n");
    }
}
bool isSafe(int board[MAX][MAX],int row,int col,int N){
    //checking column
    for(int i=0;i<row;i++){
        if(board[i][col]){
            return false;
        }
    }
    //checking upper-left diagonal
    for(int i=row,j=col;i>=0 && j>=0;i--,j--){
        if(board[i][j]){
            return false;
        }
    }
    //checking upper-right diagonal
    for(int i=row,j=col;i>=0 && j<N;i--,j++){
        if(board[i][j]){
            return false;
        }
    }
    return true;
}
bool solveNQueens(int board[MAX][MAX],int row,int N){
    if(row==N){
        return true;
    }
    for(int col=0;col<N;col++){
        if(isSafe(board,row,col,N)){
            board[row][col]=1;
            if(solveNQueens(board,row+1,N)){
                return true;
            }
            board[row][col]=0;
        }
    }
    return false;
}
void solveNQueen(int N){
    for(int i=0;i<N;i++){
        for(int j=0;j<N;j++){
            board[i][j]=0;
        }
    }
    if(!solveNQueens(board,0,N)){
        printf("Solution does not exist.");
    }
    printSolution(N);
}
void main(){
    int N;
    printf("Enter the no. of queens:");
    scanf("%d",&N);
    solveNQueen(N);
}
