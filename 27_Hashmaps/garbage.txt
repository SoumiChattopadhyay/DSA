private void resize(){
        ArrayList<LinkedList<Entity>> temp = new ArrayList<>(entities.size()*2);
        temp.addAll(entities);
        entities=temp;
    }

            LinkedList<Entity> subEntities = new LinkedList<>();
            subEntities.add(en);
            entities.set(hash,subEntities);
        }
        else{//if at idx=hash arraylist already has a linkedlist then get the linkedlist and traverse the linkedist to see if key matches any of the keys of the entity objects then update that particular key's value(Then its an update operation not insert) otherwise if key doesnt matches then add the entity object in the list(insertion)
            LinkedList<Entity> subEntities = entities.get(hash);
            for(Entity entity:subEntities){
                if(entity.key.equals(key)){//In Java, == checks if two references point to the same object in memory, not if their contents are equal. For Strings, you should use: .equals()

                }
            }
            subEntities.add(en);
        }
